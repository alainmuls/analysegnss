121        """
122        archive_file archives the created ascii files.
123        It copies the source to the archiving destination and adds a timestamp.
124
125        args:
126        fn: file name to archive
127        dest_dir: name of archive directory
128
129        """
130        # Getting directory of file to archive
131        dir_fn = os.path.dirname(fn)
132        dir_archive_ofn = os.path.join(dir_fn, dest_dir)
133        self.logger.debug(f"archive directory of file is {dir_archive_ofn}")
134
135        # create directory if it does not exist
136        if not os.path.exists(dir_archive_ofn):
137            try:
138                os.makedirs(dir_archive_ofn, exist_ok=True)
139                self.logger.info(f"Directory {dir_archive_ofn} created.")
140            except Exception as e:
141                self.logger.error(
142                    f"Error creating archive directory {dir_archive_ofn}: {e}"
143                )
144
145        # extract base name
146        fn_base = os.path.basename(fn)
147
148        # Get timestamp
149        timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M")
150        fn_time = timestamp + "_" + fn_base
151
152        # update destination
153        dest = os.path.join(dir_archive_ofn, fn_time)
154
155        # copy file to archive directory
156        try:
157            shutil.copy2(fn, dest)
158            self.logger.info(f"Successfully archived file {fn} to {dest}")
159
160        except Exception as e:
161            self.logger.error(
162                f"Error moving file {fn} to archive directory {dest}: {e}"

170        def clean_block_name(text):

172            text = text.replace(" (v1)", "")
173            # Convert " (v2)" to "2"

175            return text

177        # sbf to CSV conversion utility
178        run_sbfblocks = locate("sbfblocks")
179        # options for sbfblocks

501    def add_columns(self, block_df: pl.DataFrame) -> pl.DataFrame:
502        """checks if we can create a datetime,PRN, UTM columns in the dataframe
503
504        Args:
505                block_df (pl.DataFrame): dataframe corresponding to a SBF block
506
507        Returns:
508                pl.DataFrame: dataframe with datetime and PRN columns added if possible
509        """
510        # print(f"block_df = \n{block_df}")
511        # remove the rows where 'Type' equals 0 (no PVT available)
512        if self.logger:
513            self.logger.debug("\tremoving rows with no PVT solution")
514
515        if "Type" in block_df.columns:
516            block_df = block_df.filter(pl.col("Type") != 0).lazy()
517            # print(f"block_df = \n{block_df}")
518
519        # add date-time and PRN (as str) to the dataframe
520        if "WNc [w]" in block_df.columns and "TOW [0.001 s]" in block_df.columns:
521            if self.logger:
522                self.logger.debug("\tadding datetime column to the dataframe")
523            block_df = block_df.with_columns(
524                pl.struct(["WNc [w]", "TOW [0.001 s]"])
525                .map_elements(
526                    lambda x: gpsms2dt(week=x["WNc [w]"], towms=x["TOW [0.001 s]"]),
527                    return_dtype=pl.Datetime,
528                )
529                .alias("DT")
530            ).lazy()
531
532        # add date-time and PRN (as str) to the dataframe
533        if "WNc [w]" in block_df.columns and "TOW [s]" in block_df.columns:
534            if self.logger:
535                self.logger.debug("\tadding datetime column to the dataframe")
536            block_df = block_df.with_columns(
537                pl.struct(["WNc [w]", "TOW [s]"])
538                .map_elements(
539                    lambda x: gpsms2dt(week=x["WNc [w]"], towms=x["TOW [s]"]),
540                    return_dtype=pl.Datetime,
541                )
542                .alias("DT")
543            ).lazy()
544
545        # add date-time and PRN (as str) to the dataframe
546        if (
547            "SVID"
548            in block_df.columns
549            # and not block_df.select(pl.col("SVID")).dtypes[0] == pl.String
550        ):
551            if self.logger:
552                self.logger.debug("\tadding PRN column to the dataframe")
553            block_df = block_df.with_columns(
554                pl.struct(["SVID"])
555                .map_elements(
556                    lambda x: sbfc.ssnerr_prn2str(prn=x["SVID"]),
557                    return_dtype=pl.Utf8,
558                )
559                .alias("PRN")
560            ).lazy()
561            block_df = block_df.drop(["SVID"]).lazy()
562        # else:
563        #     # rename the column to PRN
564        #     block_df = block_df.rename({"SVID": "PRN"}).lazy()
565
566        # add UTM coordinates
567        if (
568            "Latitude [rad]" in block_df.columns
569            and "Longitude [rad]" in block_df.columns
570        ):
571            if self.logger:
572                self.logger.debug("\tadding UTM coordinates to the dataframe")
573
574            # Function to convert lat/lon in degrees to UTM
575            def latlon_to_utm(lat, lon):
576                easting, northing, _, _ = utm.from_latlon(lat, lon)
577                return {"easting": easting, "northing": northing}
578
579            # Convert latitude and longitude from radians to degrees
580            block_df = block_df.with_columns(
581                [
582                    (pl.col("Latitude [rad]") * 180 / np.pi).alias("latitude [deg]"),
583                    (pl.col("Longitude [rad]") * 180 / np.pi).alias("longitude [deg]"),
584                ]
585            ).lazy()
586
587            # Apply the conversion function lazily using map_elements with specified return_dtype
588            block_df = block_df.with_columns(
589                [
590                    pl.struct(["latitude [deg]", "longitude [deg]"])
591                    .map_elements(
592                        lambda row: latlon_to_utm(
593                            row["latitude [deg]"], row["longitude [deg]"]
594                        ),
595                        return_dtype=pl.Struct(
596                            [
597                                pl.Field("easting", pl.Float64),
598                                pl.Field("northing", pl.Float64),
599                            ]
600                        ),
601                    )
602                    .alias("utm_coords")
603                ]
604            ).lazy()
605
606            # Extract the UTM.East and UTM.North from the computed struct
607            block_df = block_df.with_columns(
608                [
609                    pl.col("utm_coords").struct.field("easting").alias("UTM.E"),
610                    pl.col("utm_coords").struct.field("northing").alias("UTM.N"),
611                ]
612            ).lazy()
613
614            # Drop intermediate columns
615            block_df = block_df.drop(
616                ["Latitude [rad]", "Longitude [rad]", "utm_coords"]
617            ).lazy()
618
619        # add orthometric height to the dataframe
620        if "Height [m]" in block_df.columns and "Undulation [m]" in block_df.columns:
621            if self.logger:
622                self.logger.debug("\tadding orthometric height to the dataframe")
623            block_df = block_df.with_columns(
624                pl.struct(["Height [m]", "Undulation [m]"])
625                .apply(
626                    lambda x: x["Height [m]"] - x["Undulation [m]"],
